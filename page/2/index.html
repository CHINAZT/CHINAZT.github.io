    //    1）获取要操作的元素
    //    2）给元素绑定一个点击的事件
    //    3）执行换颜色的操作

    document.getElementById('元素的Id值');

    //    get  获取
    //    Element 一个元素
    //    By  通过
    //    Id  元素的id值

    //    在整个文档中，通过元素的id获取一个元素

    //    2）给元素绑定一个点击的事件
    //    元素.onclick = function(){
    //        点击时执行的代码
    //    }

//        3）执行换颜色的操作
        //获取当前的背景颜色
//        只能获取行内样式的css属性
//        background-color  --->backgroundColor
//        margin-left  ---->marginLeft

        var bg = oBody.style.backgroundColor;
        /* if(bg=='white'){
         oBody.style.backgroundColor = 'black';

         oBody.style.backgroundColor = 'white';
         }*/

//        bg=='white'?oBody.style.backgroundColor = 'black':oBody.style.backgroundColor = 'white';

//this当前点击的元素 ，代表oBody
        switch (bg) {
            case 'white':
                this.style.backgroundColor = 'red';
                break;

                break;

                break;

                this.style.backgroundColor = 'green';
                break;

                break;

        }

     }else if(条件2){
     执行条件2成立的代码
     }else if(条件3){
     执行条件3成立的代码
     }....else{

     }*/

    var a = 12;
    if (a < 0) {
        console.log('我是一个负数')
    } else if (a >= 0 || a <= 10) {
        console.log('我是0到10之间的数')
    } else {
        console.log('我是大于10的数')
    }

    var b = 10;
    if (b <= 0) {
        console.log('负数或者0');
    } else {
        console.log('正数')
    }

    b <= 0 ? console.log('负数或者0') : console.log('正数');

    //if的条件中也可以有一个值得情况，直接判断这个值是真还是假
    if (0) {
        alert('张三');
    }
    //void 0  用来补位的，没有其他的意义
    0 ? alert('张三') : void 0;

    /*switch (要判断的值){

     case 值2:
     条件成立执行的代码;
     break;

     default:
     以上条件都不成立执行的代码

     }*/


    var a = 1;

     console.log('今天星期一');
     }else if(a==2){
     console.log('今天星期二');

     }else{

1-1.8boolean 

<!DOCTYPE html>
<html>
<head lang="en">
    <meta charset="UTF-8">
    <title></title>
</head>
<body>
<script>
    //boolean  布尔数据类型：只有两个值 true false
    //!  取反
    console.log(!3);  //false
    //    首先将3转换为boolean数据类型  执行Boolean(3)--》true的方法，然后再取反  --》false

    //    !要检测的值  先默认将这个值转换boolean类型，然后再取反
    //    如果一个值转换为布尔数据类型，只有0、NaN、''、null、undefined是false，其余都是true

    //    !! 先取反再取反，相当于直接转换为布尔数据类型
    //    Boolean(要转换的内容)  将其他数据类型转换为布尔数据类型


    //    Boolean(3)  -->true
    //    !3 --->false
    //    !!3 --->true

    //    = 赋值
    //    == 比较
    //
    //    1==1  -- true
    //    1==2   --false
    //    "zhang"=="zhangsan"  --false
    //    "zs"=="zs"   --true
    //    相同数据类型的两个值直接比较是否相等

    //    val1==val2
    //    不同数据类型的两个值之间的比较，遵循以下规则
    /*
     转换为数字  默认调用Number()
     转换为字符串  默认调用toString()
     **布尔转换为数字 true 1   false 0

     1）对象==对象  永远不相等
     []==[]  false
     {}=={}  false
     2）对象==字符串  先将对象转换为字符串，然后再进行比较
     []==''
     首先，[]转化为字符串，浏览器会默认执行toString()的方法 ==[].toString()  -->""  然后，""==''  比较，返回true

     3）对象==布尔  对象先转换字符串，再转换为数字；布尔也转换为数字
     []==false  -->true
     [].toString() ---""   Number("") ---0
     Number(false)  ---0

     4)对象==数字    对象先转字符串，再转换为数字
     []==0  -->true
     [].toString() ---''   Number('')---0

     5)布尔==字符串   都转还为数字
     true=='0'  false
     Number(true) ---1
     Number('0') ---0
     6)布尔==数字     布尔转化为数字
     true==1 --->true   Number(true)  1
     7)数字==字符串   字符串转化为数字
     1=='123'
     8)null==undefined  true
     9)null或者undefined和其他任何数据类型的值比较，永远不相等


     */

    //    =赋值
    //    ==比较
    //    ===严格比较：数据类型和值必须全部都相等 返回true，如果数据类型不相等 返回false

    //    null==undefined     --true
    //    null===undefined    --false


    //    Boolean 只有两个值：true false
    //    !     取反
    //    !!
    //    Boolean()
    //    0、NaN、''、null、undefined  是false


</script>

</body>
</html>

//    isNaN();   是不是一个数，判断一个值是不是有效数字，如果不是一个有效数字返回true，如果是一个有效数字返回false
//    isNaN(要判断的值);  返回值：true false

    console.log(isNaN(123));  //false
    console.log(isNaN('zhangsan')); //true
    console.log(isNaN('123'));
//    首先，将字符串（'123'）转化为数字--->123
//    然后，在进行isNaN(123)   --->false

//    isNaN();  要检测值如果是字符串，浏览器默认的先将字符串转化为数字，再进行判断


//    强制转换为number类型：Number();
//    Number(); 将其他数据类型转换为number数据类型
    console.log(Number('123'));
//    Number('123px');  -->NaN
//    Number(要转换的内容) 如果要转换的内容是字符串，字符串中必须全部都是数字，才可以转换，否则返回（结果）NaN

//    非强制转换：parseInt  parseFloat

    console.log(parseInt('12px')); //12
//    parseInt(要转换的内容):从左到右依次查找，如果是数字 返回和数字，中途就如果遇到一个不是数字的字符，就不再继续查找了，直接返回之前的数字，如果要转换的内容第一个字符就不是数字，直接返回NaN

    console.log(parseInt('px123z'));//NaN

//    parseFloat：和parseInt一样，只不过可以多识别一个点.
    console.log(parseFloat('123.5px')); //123.5
//    parseFloat('12.3.4asd')  -->12.3

//    =赋值
//    ==比较
//    1==1  true（真）  false（假）

//    NaN==NaN  返回（结果）false
//    Number('')   0
//    Number(null)  0

    var num = Number('12px');//num = NaN
    if(num==12){
        console.log('12');
    }else if(num==NaN){
        console.log('NaN');
    }else{
        console.log('以上都不成立')
    }
//    number：正数、负数、小数、0、NaN

//    isNaN();  判断

//    Number()
