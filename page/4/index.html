<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<script>
    //函数  function
    //一个功能  一个方法

    //求和的方法
    //函数分为两部分：1，创建函数；2，执行函数
    //创建函数
    function sum() {
        //函数体:实现这个方法，执行的代码
        var total = null;
        total = 1 + 1;
        console.log(total);
    }

    sum();//执行函数
    sum();//一个函数可以执行很多次，每次执行和上一次执行没有任何关系
    sum;

    //函数执行的时候会形成一个私有作用域，函数体里面的代码从上到下执行，外界无法访问，也无法修改，我们把函数执行时的这种机制叫做闭包。闭包不是一种形式（书写），而是一种机制


</script>

<script>
    //思路：
    //    1）获取要操作的元素
    //    2）给元素绑定一个点击的事件
    //    3）执行换颜色的操作

    document.getElementById('元素的Id值');
    //    document  文档 （上下文，范围）
    //    get  获取
    //    Element 一个元素
    //    By  通过
    //    Id  元素的id值

    //    在整个文档中，通过元素的id获取一个元素

    //    1）获取要操作的元素
    var oBody = document.getElementById('oBody');
    //    2）给元素绑定一个点击的事件
    //    元素.onclick = function(){
    //        点击时执行的代码
    //    }
    oBody.onclick = function () {
//        3）执行换颜色的操作
        //获取当前的背景颜色
//        只能获取行内样式的css属性
//        background-color  --->backgroundColor
//        margin-left  ---->marginLeft
//        alert(oBody.style.backgroundColor);
        var bg = oBody.style.backgroundColor;
        /* if(bg=='white'){
         oBody.style.backgroundColor = 'black';
         }else{
         oBody.style.backgroundColor = 'white';
         }*/

//        bg=='white'?oBody.style.backgroundColor = 'black':oBody.style.backgroundColor = 'white';

//this当前点击的元素 ，代表oBody
        switch (bg) {
            case 'white':
                this.style.backgroundColor = 'red';
                break;
            case 'red':
                this.style.backgroundColor = 'orange';
                break;
            case 'orange':
                this.style.backgroundColor = 'yellow';
                break;
            case 'yellow':
                this.style.backgroundColor = 'green';
                break;
            case 'green':
                this.style.backgroundColor = 'blue';
                break;
            default :
                this.style.backgroundColor = 'white';
        }


    }


</script>

</body>

    <!DOCTYPE html>

<script>
    //    if、else if、else
    //    三元运算符（三目运算符）
    //    switch case

    //    if else
    /*if(条件1){
     执行条件1成立的代码
     }else if(条件2){
     执行条件2成立的代码
     }else if(条件3){
     执行条件3成立的代码
     }....else{
     以上条件都不成立执行的代码
     }*/


    //if的条件中可以写大于,小于，等于
    //&&  并且 两边的条件必须全都成立 整体才为真
    //||  或者 两边的小条件，只有一个为真，整体即为真
    var a = 12;
    if (a < 0) {
        console.log('我是一个负数')
    } else if (a >= 0 || a <= 10) {
        console.log('我是0到10之间的数')
    } else {
        console.log('我是大于10的数')
    }

    //    三元运算符:应用于简单的if else的形式  或者只有if的情况
    var b = 10;
    if (b <= 0) {
        console.log('负数或者0');
    } else {
        console.log('正数')
    }

    //    判断的条件?条件成立执行的代码:j条件不成立执行的代码
    b <= 0 ? console.log('负数或者0') : console.log('正数');

    //if的条件中也可以有一个值得情况，直接判断这个值是真还是假
    if (0) {
        alert('张三');
    }
    //void 0  用来补位的，没有其他的意义
    0 ? alert('张三') : void 0;

    //    switch case：用于不同值的情况
    /*switch (要判断的值){
     case 值1:
     条件成立执行的代码;
     break;
     case 值2:
     条件成立执行的代码;
     break;
     ...
     default:
     以上条件都不成立执行的代码

     }*/


    var a = 1;
    /*if(a==1){
     console.log('今天星期一');
     }else if(a==2){
     console.log('今天星期二');
     }else if(a==3){
     console.log('今天星期三');
     }else if(a==4){
     console.log('今天星期四');
     }else{
     console.log('今天放假')
     }*/
    //每个case后面都要加break，如果不加break，不管后面的条件是否成立都会执行
    //每个case的情况是===的比较，严格比较

    switch (a) {
        case 1:
            console.log('今天星期一');
            break;
        case 2:
            console.log('今天星期二');
            break;
        case 3:
            console.log('今天星期三');
            break;
        case 4:
            console.log('今天星期四');
            break;
        default :
            console.log('今天放假');

    }


</script>
