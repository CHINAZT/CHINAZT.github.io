2-2.5任意个参数求和 

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title></title>
</head>
<body>
<!--
思路：add()
参数：没有
返回值：把所有参数求和后的结果
1）遍历arguments；
2）判断每个arguments是否为有效数字，如果是有效数字，进行累加；注意累加的时候，预防字符串拼接的问题；
-->
<script>
    function add() {
        var total = 0;
        //遍历每个arguments，判断每个arguments是否为有效数字
        for (var i = 0; i < arguments.length; i++) {
            if (!isNaN(arguments[i])) {
                total += Number(arguments[i]);//为了防止字符串拼接；
            }
        }
        return total;//最后把求和后的结果返回；
    }
    console.log(add(12, 32, 9, '3', '3px'))
</script>
</body>

2-2.4上一个哥哥元素节点 

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title></title>
</head>
<body>
<ul>
    <li>0</li>
    <li>1</li>
    <li>2</li>
    <li>3</li>
</ul>
<script>
    var aLi = document.getElementsByTagName('li');
    //prev:找当前元素的上一个哥哥元素节点；
    console.log(prev(aLi[2]).innerHTML)
    function prev(curEle) {//curEle形参
        //兼容高级浏览器；previousElementSibling：直接拿到上一个哥哥元素节点；
        if (curEle.previousElementSibling) {
            return curEle.previousElementSibling;
        }
        //处理低级浏览器的兼容问题
        var pre = curEle.previousSibling;//获取上一个哥哥节点；
        //上一个节点存在，并且不是元素节点；才会继续往上找；
        while (pre && pre.nodeType !== 1) {
            pre = pre.previousSibling;
        }
        return pre;
    }
</script>
</body>
</html>
