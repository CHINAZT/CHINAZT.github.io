    //思路：
    //    1）获取要操作的元素
    //    2）给元素绑定一个点击的事件
    //    3）执行换颜色的操作

    document.getElementById('元素的Id值');
    //    document  文档 （上下文，范围）
    //    get  获取
    //    Element 一个元素
    //    By  通过
    //    Id  元素的id值

    //    元素.onclick = function(){

//        3）执行换颜色的操作
        //获取当前的背景颜色
//        只能获取行内样式的css属性
//        background-color  --->backgroundColor

        switch (bg) {
            case 'white':
                this.style.backgroundColor = 'red';

    //    三元运算符（三目运算符）
    //    switch case

    //    if else
    /*if(条件1){
     执行条件1成立的代码
     }else if(条件2){
     执行条件2成立的代码
     }else if(条件3){
     执行条件3成立的代码
     }....else{
     以上条件都不成立执行的代码
     }*/


    //if的条件中可以写大于,小于，等于
    //&&  并且 两边的条件必须全都成立 整体才为真
    //||  或者 两边的小条件，只有一个为真，整体即为真
    var a = 12;
    if (a < 0) {
        console.log('我是一个负数')
    } else if (a >= 0 || a <= 10) {
        console.log('我是0到10之间的数')
    } else {
        console.log('我是大于10的数')
    }

    //    三元运算符:应用于简单的if else的形式  或者只有if的情况
    var b = 10;
    if (b <= 0) {
        console.log('负数或者0');
    } else {
        console.log('正数')
    }

    //    判断的条件?条件成立执行的代码:j条件不成立执行的代码
    b <= 0 ? console.log('负数或者0') : console.log('正数');

    //if的条件中也可以有一个值得情况，直接判断这个值是真还是假
    if (0) {
        alert('张三');
    }
    //void 0  用来补位的，没有其他的意义
    0 ? alert('张三') : void 0;

    //    switch case：用于不同值的情况
    /*switch (要判断的值){
     case 值1:
     条件成立执行的代码;
     break;
     case 值2:
     条件成立执行的代码;
     break;
     ...
     default:
     以上条件都不成立执行的代码

     }*/


    var a = 1;
    /*if(a==1){
     console.log('今天星期一');
     }else if(a==2){
     console.log('今天星期二');
     }else if(a==3){
     console.log('今天星期三');
     }else if(a==4){
     console.log('今天星期四');
     }else{
     console.log('今天放假')
     }*/
