<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
</head>
<body>
<script>
    //函数  function
    //一个功能  一个方法

    //求和的方法
    //函数分为两部分：1，创建函数；2，执行函数
    //创建函数
    function sum() {
        //函数体:实现这个方法，执行的代码
        var total = null;
        total = 1 + 1;
        console.log(total);
    }

    sum();//执行函数
    sum();//一个函数可以执行很多次，每次执行和上一次执行没有任何关系
    sum;

    //函数执行的时候会形成一个私有作用域，函数体里面的代码从上到下执行，外界无法访问，也无法修改，我们把函数执行时的这种机制叫做闭包。闭包不是一种形式（书写），而是一种机制


</script>
